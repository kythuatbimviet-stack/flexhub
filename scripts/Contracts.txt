/**
 * @param {Object|string} dataJson - Dữ liệu placeholder (JSON object hoặc string)
 */
function generateContractFromJSON(dataJson) {
  try {
    // 1. Parse dữ liệu nếu là chuỗi
    const payload = typeof dataJson === 'string' ? JSON.parse(dataJson) : dataJson;
    
    // Lấy ID hợp đồng để đặt tên file và update Supabase (Giả định key là {{id}})
    const contractId = payload["{{id}}"] || "N/A";
    const fileName = "Hợp đồng_" + contractId + "_" + new Date().getTime();

    // 2. Tạo bản sao từ mẫu
    const templateFile = DriveApp.getFileById(CONFIG.TEMPLATE_ID);
    const targetFolder = DriveApp.getFolderById(CONFIG.FOLDER_ID);
    const tempDocFile = templateFile.makeCopy(fileName, targetFolder);
    const doc = DocumentApp.openById(tempDocFile.getId());
    const body = doc.getBody();

    // 3. Duyệt qua JSON để thay thế placeholder
    for (let key in payload) {
      let value = payload[key] || "....................";
      
      // Xử lý định dạng nếu là Date
      if (value instanceof Date) {
        value = Utilities.formatDate(value, Session.getScriptTimeZone(), "dd/MM/yyyy");
      }

      // A. Thay thế dạng cơ bản: {{key}}
      body.replaceText(escapeRegex(key), value.toString());

      // B. Xử lý các biến thể tự động (UPPER và TEXT)
      // Lấy tên gốc (ví dụ: {{member_name}} -> member_name)
      const rawKey = key.replace(/[{}]/g, ""); 
      
      // Thay thế {{UPPER(key)}}
      body.replaceText("{{UPPER\\(" + rawKey + "\\)}}", value.toString().toUpperCase());

      // Thay thế {{TEXT(key)}} - Định dạng số việt nam
      let formattedValue = value;
      if (!isNaN(parseFloat(value)) && isFinite(value)) {
          formattedValue = Number(value).toLocaleString('vi-VN');
      }
      body.replaceText("{{TEXT\\(" + rawKey + "\\)}}", formattedValue.toString());
    }

    doc.saveAndClose();

    // 4. Xuất PDF và xóa file tạm
    const pdfBlob = tempDocFile.getAs(MimeType.PDF);
    const pdfFile = targetFolder.createFile(pdfBlob).setName(fileName + ".pdf");
    const pdfUrl = pdfFile.getUrl();
    tempDocFile.setTrashed(true);

    // 5. Cập nhật vào Supabase
    updateSupabase(contractId, pdfUrl);

    console.log("Thành công! Link PDF: " + pdfUrl);
    return pdfUrl;

  } catch (e) {
    console.error("Lỗi: " + e.toString());
    return "Lỗi: " + e.message;
  }
}

/**
 * Hàm cập nhật link PDF vào Supabase
 */
function updateSupabase(id, url) {
  const endpoint = `${CONFIG.SUPABASE_URL}/rest/v1/${CONFIG.TABLE_NAME}?id=eq.${id}`;
  
  const options = {
    method: "patch",
    headers: {
      "apikey": CONFIG.SUPABASE_KEY,
      "Authorization": "Bearer " + CONFIG.SUPABASE_KEY,
      "Content-Type": "application/json",
      "Prefer": "return=minimal"
    },
    payload: JSON.stringify({
      contract_file_url: url,
      updated_at: new Date().toISOString()
    }),
    muteHttpExceptions: true
  };

  const response = UrlFetchApp.fetch(endpoint, options);
  
  if (response.getResponseCode() !== 204 && response.getResponseCode() !== 200) {
    throw new Error("Lỗi Supabase: " + response.getContentText());
  }
  Logger.log("Đã cập nhật Supabase cho ID: " + id);
}

/**
 * Hàm bổ trợ để xử lý các ký tự đặc biệt trong Regex
 */
function escapeRegex(string) {
  return string.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');
}

function mapDataToContract(rowData) {
  const payload = {};

  // 1. Duyệt qua toàn bộ các cột có sẵn trong dữ liệu gốc
  for (let key in rowData) {
    let value = rowData[key] || ""; // Tránh null/undefined

    // Xử lý định dạng ngày tháng (YYYY-MM-DD -> DD/MM/YYYY)
    if (key.includes('date') || key === 'dob' || key === 'signing_date') {
      value = formatDateVietnamese(value);
    }

    // A. Map mặc định: {{tên_cột}}
    payload["{{" + key + "}}"] = value;

    // B. Map các trường bị lệch tên (Fix lỗi bạn nêu)
    if (key === 'short_name') payload["{{center_short_name}}"] = value;
    if (key === 'facility_name') payload["{{center_name}}"] = value;
    if (key === 'member_address') payload["{{address}}"] = value; // Địa chỉ hội viên
    
    // C. Tự động tạo định dạng IN HOA cho tất cả các trường
    payload["{{UPPER(" + key + ")}}"] = value.toString().toUpperCase();

    // D. Tự động tạo định dạng Tiền tệ & Chữ cho các trường số tiền
    if (key.includes('price') || key.includes('amount')) {
       let num = parseFloat(value) || 0;
       payload["{{TEXT(" + key + ")}}"] = num.toLocaleString('vi-VN');
       payload["{{" + key + "_words}}"] = docSoTien(num);
    }
  }

  // 2. Bổ sung các giá trị cố định nếu Database bị thiếu (Fallback)
  if (!payload["{{bank_name}}"]) payload["{{bank_name}}"] = "MB Bank";
  if (!payload["{{account_number}}"]) payload["{{account_number}}"] = "1990231299999";

  return payload;
}
function formatDateVietnamese(dateStr) {
  if (!dateStr) return "....................";
  try {
    const d = new Date(dateStr);
    return Utilities.formatDate(d, Session.getScriptTimeZone(), "dd/MM/yyyy");
  } catch(e) { return dateStr; }
}

/**
 * HÀM CHÍNH: Xử lý tạo file và đồng bộ Supabase
 */
function processContractJSON(data) {
  try {
    const contractId = data["{{id}}"];
    if (!contractId) throw new Error("Dữ liệu JSON thiếu key {{id}}");

    const fileName = "Hợp đồng_" + contractId;
    
    // 1. Tạo file Doc tạm từ mẫu
    const template = DriveApp.getFileById(CONFIG.TEMPLATE_ID);
    const folder = DriveApp.getFolderById(CONFIG.FOLDER_ID);
    const tempFile = template.makeCopy(fileName, folder);
    const doc = DocumentApp.openById(tempFile.getId());
    const body = doc.getBody();

    // 2. Duyệt qua từng Key-Value trong JSON để thay thế
    for (let key in data) {
      let val = data[key] === null || data[key] === "" ? "...................." : data[key];
      
      // Xử lý nếu là dữ liệu ngày tháng
      if (val instanceof Date) {
        val = Utilities.formatDate(val, Session.getScriptTimeZone(), "dd/MM/yyyy");
      }

      // Lấy tên key không có dấu ngoặc (ví dụ: member_name)
      const rawKey = key.replace(/[{}]/g, "");

      // A. Thay thế dạng thường: {{member_name}}
      body.replaceText(escapeRegex(key), val.toString());

      // B. Thay thế dạng UPPER: {{UPPER(member_name)}}
      body.replaceText("{{UPPER\\(" + rawKey + "\\)}}", val.toString().toUpperCase());

      // C. Thay thế dạng Tiền tệ: {{TEXT(total_amount)}}
      let currencyVal = val;
      if (!isNaN(parseFloat(val)) && isFinite(val)) {
        currencyVal = Number(val).toLocaleString('vi-VN');
      }
      body.replaceText("{{TEXT\\(" + rawKey + "\\)}}", currencyVal.toString());
    }

    doc.saveAndClose();

    // 3. Chuyển sang PDF
    const pdfBlob = tempFile.getAs(MimeType.PDF);
    const pdfFile = folder.createFile(pdfBlob).setName(fileName + ".pdf");
    const pdfUrl = pdfFile.getUrl();

    // 4. Xóa file Doc tạm
    tempFile.setTrashed(true);

    // 5. Ghi ngược vào Supabase
    updateSupabase(contractId, pdfUrl);

    return pdfUrl;

  } catch (e) {
    Logger.log("LỖI: " + e.message);
    throw e;
  }
}
